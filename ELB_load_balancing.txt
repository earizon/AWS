######################################
## Create application load balancer ##
######################################

VPC_NAME="myvpc"
# STEP 1.1) Create VPC
AWS_VPC_ID=$(aws ec2 create-vpc \
   --cidr-block 10.0.0.0/16 \
   --query 'Vpc.{VpcId:VpcId}' \
   --output text)

# STEP 1.2) Enable DNS-hostname for the VPC
aws ec2 modify-vpc-attribute \
--vpc-id $AWS_VPC_ID \
--enable-dns-hostnames "{\"Value\":true}"

# STEP 1.3) "tag" to VPC
aws ec2 create-tags \
--resources $AWS_VPC_ID \
--tags "Key=Name,Value=${VPC_NAME}"

# STEP 2) Create an Internet Gateway
AWS_INTERNET_GATEWAY_ID=$(
  aws ec2 create-internet-gateway \
--query 'InternetGateway.{InternetGatewayId:InternetGatewayId}' \
--output text )

# Attach Internet-gateway to VPC
aws ec2 attach-internet-gateway \
--vpc-id $AWS_VPC_ID \
--internet-gateway-id $AWS_INTERNET_GATEWAY_ID

# Add a tag to the Internet-Gateway
aws ec2 create-tags \
--resources $AWS_INTERNET_GATEWAY_ID \
--tags "Key=Name,Value=${VPC_NAME}-internet-gateway"

#  Create a route-table
AWS_CUSTOM_ROUTE_TABLE_ID=$(
  aws ec2 create-route-table \
    --vpc-id $AWS_VPC_ID \
    --query 'RouteTable.{RouteTableId:RouteTableId}' \
    --output text )

# Create route to Internet Gateway
aws ec2 create-route \
--route-table-id $AWS_CUSTOM_ROUTE_TABLE_ID \
--destination-cidr-block 0.0.0.0/0 \
--gateway-id $AWS_INTERNET_GATEWAY_ID


for CIDR in 10.0.1.0/24 10.0.2.0/24 ; do # {
  # NEXT) Create two public subnets
  AWS_SUBNET_PUBLIC_ID=$(
    aws ec2 create-subnet \
      --vpc-id $AWS_VPC_ID \
      --cidr-block $CIDR \
      --availability-zone ${AWS_REGION1} \
      --query 'Subnet.{SubnetId:SubnetId}' \
      --output text )

  # NEXT) Enable Auto-assign Public IP on Public Subnets
  aws ec2 modify-subnet-attribute \
    --subnet-id $AWS_SUBNET_PUBLIC_ID \
    --map-public-ip-on-launch

  # NEXT) "tag" public subnets
  TAG_VALUE=$(echo $CIDR | sed "s/\//_/g" | sed "s/\./_/g")
  aws ec2 create-tags \
  --resources $AWS_SUBNET_PUBLIC_ID \
  --tags "Key=Name,Value=${TAG_VALUE}"

  ## Associate public-subnet with route-table
  AWS_ROUTE_TABLE_ASSOID_ONE=$(aws ec2 associate-route-table  \
    --subnet-id $AWS_SUBNET_PUBLIC_ID \
    --route-table-id $AWS_CUSTOM_ROUTE_TABLE_ID \
    --query 'AssociationId' \
    --output text )
done  # }

# Create a security group
aws ec2 create-security-group \
  --vpc-id $AWS_VPC_ID \
  --group-name ${VPC_NAME}-security-group \
  --description 'My VPC non default security group'

## Get security group ID's
AWS_DEFAULT_SECURITY_GROUP_ID=$(
  aws ec2 describe-security-groups \
    --filters "Name=vpc-id,Values=$AWS_VPC_ID" \
    --query 'SecurityGroups[?GroupName == `default`].GroupId' \
    --output text) &&
AWS_CUSTOM_SECURITY_GROUP_ID=$(
  aws ec2 describe-security-groups \
    --filters "Name=vpc-id,Values=$AWS_VPC_ID" \
    --query 'SecurityGroups[?GroupName == `${VPC_NAME}-security-group`].GroupId' \
    --output text)

## Create security group ingress rules for TLS (port 443)
aws ec2 authorize-security-group-ingress \
  --group-id $AWS_CUSTOM_SECURITY_GROUP_ID \
  --ip-permissions  '[{"IpProtocol": "tcp", "FromPort": 443, "ToPort": 443, "IpRanges": [{"CidrIp": "0.0.0.0/0", "Description": "Allow TLS"}]}]'
